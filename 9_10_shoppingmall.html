  • 문제: onclick시 calc(this)메소드 부분을 완성하도록 한다.

    • <form>
    <input type="checkbox" name="hap" value="10000" onclick="calc(this)"/>모자 1만원
    <input type="checkbox" name="shose" value="30000" onclick="calc(this)"/>구두 3만원
    <input type="checkbox" name="bag" value="80000" onclick="calc(this)"/>명품가방 8만원
    지불하실 금액 <input type="text" id="sumtext" value="0" />
    </form>

  • var sum=0;
    var submit = document.getElementById("sumtext");

    function calc(obj) {
      if(obj.checked)
        sum += parseInt(obj.value);
      else
        sum -= parseInt(obj.value);
        submit.value = sum;
        }
